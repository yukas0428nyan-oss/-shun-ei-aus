let currentMode = 'all'; let currentProblem = null; let isRunning = false; let timerInterval = null; let currentStage = 'idle'; let sessionStats = { wrong: 0, correct: 0, total: 0 }; let weakProblems = {}; // ====================== // 初期化 // ====================== window.addEventListener('DOMContentLoaded', () => { loadWeakProblems(); updateStatsDOM(); }); // ====================== // ローカル保存 // ====================== function loadWeakProblems() { const data = localStorage.getItem('aus_weak_problems'); weakProblems = data ? JSON.parse(data) : {}; } function saveWeakProblems() { localStorage.setItem('aus_weak_problems', JSON.stringify(weakProblems)); updateStatsDOM(); } // ====================== // UI更新 // ====================== function updateStatsDOM() { document.getElementById('statWrong').innerText = sessionStats.wrong; document.getElementById('statCorrect').innerText = sessionStats.correct; document.getElementById('statTotal').innerText = sessionStats.total; document.getElementById('statWeakCount').innerText = Object.keys(weakProblems).length; } // ====================== // スタート // ====================== function toggleAppLoop() { if (isRunning) { stopAppLoop(); } else { startAppLoop(); } } function startAppLoop() { isRunning = true; document.getElementById('startStopBtn').innerText = "一時停止"; nextTurn(); } function stopAppLoop() { isRunning = false; currentStage = 'idle'; if (timerInterval) clearInterval(timerInterval); window.speechSynthesis?.cancel(); } // ====================== // 問題選択 // ====================== function nextTurn() { if (!isRunning) return; let pool = quizData; if (currentMode === 'weak') { const ids = Object.keys(weakProblems).map(Number); pool = quizData.filter(q => ids.includes(q.id)); if (pool.length === 0) { alert("苦手問題なし！"); changeMode('all'); return; } } currentProblem = pool[Math.floor(Math.random() * pool.length)]; sessionStats.total++; updateStatsDOM(); runJaStage(); } // ====================== // 日本語ステージ（8〜10秒固定） // ====================== function runJaStage() { currentStage = 'ja'; document.getElementById('jaText').innerText = currentProblem.ja; document.getElementById('enText').classList.remove('show'); document.getElementById('statusMsg').innerText = "英語を考えてください"; speak(currentProblem.ja, 'ja-JP'); setTimeout(() => { runEnStage(); }, 9000); // ←ここが8〜10秒 } // ====================== // 英語ステージ（ボタン待ち） // ====================== function runEnStage() { currentStage = 'en'; document.getElementById('enText').innerText = currentProblem.en; document.getElementById('enText').classList.add('show'); document.getElementById('statusMsg').innerText = "自己評価してください"; speak(currentProblem.en, 'en-AU'); document.getElementById('correctBtn').disabled = false; document.getElementById('wrongBtn').disabled = false; } // ====================== // 回答（ここでしか次に進まない） // ====================== function handleAnswer(isCorrect) { if (currentStage !== 'en') return; currentStage = 'waiting'; if (isCorrect) { sessionStats.correct++; if (weakProblems[currentProblem.id]) { weakProblems[currentProblem.id]++; if (weakProblems[currentProblem.id] >= 3) { delete weakProblems[currentProblem.id]; } } document.getElementById('statusMsg').innerText = "OK!"; } else { sessionStats.wrong++; weakProblems[currentProblem.id] = 0; document.getElementById('statusMsg').innerText = "復習しよう"; } saveWeakProblems(); document.getElementById('correctBtn').disabled = true; document.getElementById('wrongBtn').disabled = true; setTimeout(() => { nextTurn(); // ←ここで初めて次へ }, 1200); } // ====================== // 音声 // ====================== function speak(text, lang) { window.speechSynthesis.cancel(); const u = new SpeechSynthesisUtterance(text); u.lang = lang; u.rate = lang === 'ja-JP' ? 1 : 0.9; window.speechSynthesis.speak(u); }
