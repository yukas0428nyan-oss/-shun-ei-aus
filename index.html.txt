<script>
const q = [
{ja:"お会計お願いします", en:"Can I get the bill, please?"},
{ja:"持ち帰りで", en:"To go, please."},
{ja:"Wi-Fiありますか？", en:"Do you have Wi-Fi?"},
{ja:"トイレどこ？", en:"Where is the toilet?"},
{ja:"もう一度言ってください", en:"Could you say that again?"}
];

let i = 0;
let running = false;

function start(){
  running = true;
  document.getElementById("ok").disabled = false;
  document.getElementById("ng").disabled = false;
  next();
}

function next(){
  let data = q[Math.floor(Math.random()*q.length)];
  document.getElementById("ja").innerText = data.ja;
  document.getElementById("en").innerText = data.en;
  document.getElementById("en").style.display = "none";

  setTimeout(()=>{
    document.getElementById("en").style.display = "block";
    speak(data.en);
  }, 3000);
}

function speak(text){
  let u = new SpeechSynthesisUtterance(text);
  u.lang = "en-AU";
  speechSynthesis.speak(u);
}

function answer(){
  next();
}
